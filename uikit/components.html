				Модальные окна
			

			

				

					Скрипт
					Magnific Popup
				

				

					Стили
					source/sass/vendors/_magnific.scss
				

				

					Активируются атрибутами
				

				

					data-lightbox="inline"
					- инлайн модальное окно, применяется на тег a где href="id" его цель
				

				

					data-lightbox="ajax"
					- ajax модальное окно, применяется на тег a где href="адрес" его цель
				

				

					data-lightbox="iframe"
					- внешний iframe загружается в модальное окно, применяется на тег a где href="адрес" его цель
				

				

					data-lightbox="image"
					- еденичное изображение, применяется на тег a где href="адрес" его цель
				

				

					data-lightbox="gallery"
					- галерея изображений, применяется на тег родительский тег и делегируется всем тегам a внутри него
